This rule detects file name and class name declaration

Noncompliant Code Example

/* The file name should contain the first top-level class with the same name as it */
Class1.java // Noncompliant

public class NotClass1 { // Noncompliant
    class InnerNotClass1 {

    }
}

public class Class1 {
    class InnerClass1 {

    }
}

Compliant Solution

Class1.java // Compliant

public class Class1 { // Compliant
    class InnerClass1 {

    }
}

public class NotClass1 { 
    class InnerNotClass1 {

    }
}

or 

NotClass1.java // Compliant

public class NotClass1 { // Compliant
    class InnerNotClass1 {

    }
}

public class Class1 {
    class InnerClass1 {

    }
}
